async function analyzeVideo() { const urlInput = document.getElementById('videoUrl').value; const resultArea = document.getElementById('resultArea'); const loading = document.getElementById('loading'); const aiReason = document.getElementById('aiReason'); // UI Reset resultArea.classList.add('hidden'); loading.classList.remove('hidden'); try { const response = await fetch('/api/analyze', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ url: urlInput }) }); const data = await response.json(); if (data.analysis) { // Display Thumbnail document.getElementById('thumb').src = getThumbnail(urlInput); // Render the HTML table from the AI aiReason.innerHTML = data.analysis.replace(/```html|```/g, ""); resultArea.classList.remove('hidden'); } else { alert("AI Error: " + (data.error || "No response")); } } catch (err) { alert("Network Error: Make sure your Vercel deployment is green."); } finally { loading.classList.add('hidden'); } }
